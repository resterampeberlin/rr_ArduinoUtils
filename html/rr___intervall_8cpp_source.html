75Intervall::Result_t Intervall::wait(bool (*userFunc)(void)) {
76 unsigned long delta = millis() - timeStamp;
77 Result_t result = Success;
78
79#ifndef WITHOUT_INTERVALL_STATS
80 // collect statistics
81 minPeriod = min(minPeriod, delta);
82 maxPeriod = max(maxPeriod, delta);
83
84 // check for overflow
85 if (sumPeriods > ULONG_MAX - delta) {
86 PRINT_WARNING(F("Average overflow, resetting average"), NULL);
87
88 sumPeriods = getAvgPeriod() + delta;
89 numPeriods = 2;
90 }
91 else {
92 sumPeriods += delta;
93 numPeriods++;
94 }
95#endif
96
97 if (delta < period) {
98 while (millis() - timeStamp < period) {
99 if (userFunc != NULL && userFunc()) {
100 result = Abort;
101 break;
102 }
103
104 yield();
105 }
106 }
107 else {
108 PRINT_WARNING("Intervall overflow. Intervall: %u current: %u", period, delta);
109
110 result = Overflow;
111 }
112
113 timeStamp = millis();
114
115 return result;
116}
