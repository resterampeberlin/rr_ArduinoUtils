94Intervall::Result_t Intervall::wait(bool (*userFunc)(void)) {
95 Intervall::Period_t delta = millis() - timeStamp;
96 Result_t result = Success;
97
98 if (timeStamp == 0) {
99 PRINT_ERROR(F("Intervall not initialized. Call begin() before wait()"), NULL);
100 return Intervall::Failure;
101 }
102
103#ifndef WITHOUT_INTERVALL_STATS
104 // collect statistics
105 minPeriod = min(minPeriod, delta);
106 maxPeriod = max(maxPeriod, delta);
107
108 // check for overflow
109 if (sumPeriods > ULONG_MAX - delta) {
110 PRINT_WARNING(F("Average overflow, resetting average"), NULL);
111
112 sumPeriods = getAvgPeriod() + delta;
113 numPeriods = 2;
114 }
115 else {
116 sumPeriods += delta;
117 numPeriods++;
118 }
119#endif
120
121 if (delta < period) {
122 while (!isPeriodOver()) {
123 if (userFunc != NULL && userFunc()) {
124 result = Abort;
125 break;
126 }
127
128 yield();
129 }
130 }
131 else {
132 PRINT_WARNING(F("Intervall overflow. Intervall: %u current: %u"), period, delta);
133
134 result = Overflow;
135 }
136
137 timeStamp = millis();
138
139 return result;
140}
