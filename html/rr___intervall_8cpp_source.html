67Intervall::Result_t Intervall::wait(bool (*userFunc)(void)) {
68 Intervall::Period_t delta = millis() - timeStamp;
69 Result_t result = Success;
70
71 if (timeStamp == 0) {
72 PRINT_ERROR("Intervall not initialized. Call begin() before wait()", NULL);
73 return Intervall::Failure;
74 }
75
76#ifndef WITHOUT_INTERVALL_STATS
77 // collect statistics
78 minPeriod = min(minPeriod, delta);
79 maxPeriod = max(maxPeriod, delta);
80
81 // check for overflow
82 if (sumPeriods > ULONG_MAX - delta) {
83 PRINT_WARNING("Average overflow, resetting average", NULL);
84
85 sumPeriods = getAvgPeriod() + delta;
86 numPeriods = 2;
87 }
88 else {
89 sumPeriods += delta;
90 numPeriods++;
91 }
92#endif
93
94 if (delta < period) {
95 while (!isPeriodOver()) {
96 if (userFunc != NULL && userFunc()) {
97 result = Abort;
98 break;
99 }
100
101 yield();
102 }
103 }
104 else {
105 PRINT_WARNING("Intervall overflow. Intervall: %u current: %u", period, delta);
106
107 result = Overflow;
108 }
109
110 timeStamp = millis();
111
112 return result;
113}
