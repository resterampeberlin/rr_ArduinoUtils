26def generate(source, target, env):
27 if os.path.isfile(os.path.join(env['PROJECT_DIR'], "Doxyfile")):
28 template = "library-template.json"
29
30 projectBrief = ""
31 projectName = ""
32 projectNumber = gitVersion.short()
33
34 # extract information from libary-template.json
35 if os.path.isfile(template):
36 with open(template, 'r') as f:
37 data = json.load(f)
38
39 if "name" in data:
40 projectName = data["name"]
41
42 if "description" in data:
43 projectBrief = data["description"]
44
45 # feed values into doxygen
46 command = "(cat Doxyfile; "
47 command = command + "echo \"PROJECT_NUMBER=" + projectNumber + "\";"
48
49 if projectBrief:
50 command = command + "echo \"PROJECT_BRIEF=" + projectBrief + "\";"
51
52 if projectName:
53 command = command + "echo \"PROJECT_NAME=" + projectName + "\";"
54
55 command = command + " ) | doxygen -"
56
57 print("Creating documentation ...")
58
59 env.Execute(command)
60 else:
61 print("No Doxyfile found")
