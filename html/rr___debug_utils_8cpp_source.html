110bool DebugUtils::print(DebugLevel_t level, const char* location, unsigned line, const __FlashStringHelper* fmt, ...) {
111 if (shouldPrint(level) && output) {
112 va_list args;
113 char text[128];
114 String fmt_str(fmt);
115
116 // print diagnostic information
117 snprintf(text, sizeof(text), "%s %s:%d" ANSI_NORMAL "\t", getInfoMarking(level), location, line);
118 output->print(text);
119 output->print(getTextMarking(level));
120
121 // print formatted text
122 va_start(args, fmt);
123 vsnprintf(text, sizeof(text), fmt_str.c_str(), args);
124 va_end(args);
125
126 output->print(text);
127 output->println(ANSI_NORMAL);
128
129 return true;
130 }
131 else {
132 return false;
133 }
