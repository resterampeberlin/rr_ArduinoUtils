135bool DebugUtils::print(DebugLevel_t level, const char* location, unsigned line, const char* fmt, ...) {
136 if (shouldPrint(level) && output) {
137 va_list args;
138 char text[128];
139
140 // print diagnostic information
141 snprintf(text, sizeof(text), "%s %s:%d" ANSI_NORMAL "\t", getInfoMarking(level), location, line);
142
143 output->print(text);
144 output->print(getTextMarking(level));
145
146 // print formatted text
147 va_start(args, fmt);
148 vsnprintf(text, sizeof(text), fmt, args);
149 va_end(args);
150
151 output->print(text);
152 output->println(ANSI_NORMAL);
153
154 return true;
155 }
156 else {
157 return false;
158 }
159}

209void DebugUtils::setTabs(unsigned columns[], unsigned count) {
210 if (output) {
211 output->print(ANSI_CLEARTABS);
212
213 for (unsigned loop = 0, lastTab = 0; loop < count; loop++) {
214 char text[20];

218
